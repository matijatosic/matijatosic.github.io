package pomoc;

  import java.time.LocalDate;
  import java.time.format.DateTimeFormatter;

  public class KlasaClan {

  	private int id;
  	private String naziv;
  	private int celiBroj;
  	private double decimalniBroj;
  	private LocalDate datum;

  	public KlasaClan(int id, String naziv, int celiBroj, double decimalniBroj, LocalDate datum) {
  		super();
  		this.id = id;
  		this.naziv = naziv;
  		this.celiBroj = celiBroj;
  		this.decimalniBroj = decimalniBroj;
  		this.datum = datum;
  	}

  	public String getNaziv() {
  		return naziv;
  	}

  	public void setNaziv(String naziv) {
  		this.naziv = naziv;
  	}

  	public int getCeliBroj() {
  		return celiBroj;
  	}

  	public void setCeliBroj(int celiBroj) {
  		this.celiBroj = celiBroj;
  	}

  	public int getId() {
  		return id;
  	}

  	public void setId(int id) {
  		this.id = id;
  	}

  	public double getDecimalniBroj() {
  		return decimalniBroj;
  	}

  	public void setDecimalniBroj(double decimalniBroj) {
  		this.decimalniBroj = decimalniBroj;
  	}

  	public LocalDate getDatum() {
  		return datum;
  	}

  	public void setDatum(LocalDate datum) {
  		this.datum = datum;
  	}

  	public String toString() {
  		DateTimeFormatter dtf = DateTimeFormatter.ofPattern("dd.MM.yyyy.");
  		return String.format("%15s %15s %10.2f %10s %10s", id, imeVlasnika, ukupnaCena, mesec, dtf.format(datumUplate));
  	}

  }
